country_event = {
    id = 777770620
    title = "The $FROMCOUNTRY_ADJ$ Colony!"
    desc = "$FROMCOUNTRY$ is controlled by the defeated power, what should be done with it?"
	picture = "dismantle 8"
    is_triggered_only = yes
	allow_multiple_instances = yes
	
	immediate = {
		any_country = {
			limit = { 
				OR = { AND = { FROM = { OR = { government = democracy government = parliamentary_republic government = prussian_constitutionalism government = hms_government government = constitutional_empire } } THIS = { OR = { government = democracy government = parliamentary_republic government = prussian_constitutionalism government = hms_government government = constitutional_empire } } } AND = { FROM = { OR = { government = proletarian_dictatorship government = parliamentary_republic government = prussian_constitutionalism } } THIS = { OR = { government = proletarian_dictatorship } } } AND = { FROM = { OR = { government = presidential_dictatorship government = fascist_dictatorship } } THIS = { OR = { government = presidential_dictatorship government = fascist_dictatorship } } } AND = { FROM = { OR = { government = absolute_monarchy government = absolute_empire government = semi_constitutional_empire } } THIS = { OR = { government = absolute_monarchy government = absolute_empire government = semi_constitutional_empire } } } }
			}
			THIS = { set_country_flag = common_government_bonus }
		}
		any_country = {
			limit = { 
				OR = { AND = { FROM = { OR = { government = democracy government = parliamentary_republic government = prussian_constitutionalism government = hms_government government = constitutional_empire } } THIS = { OR = { government = democracy government = parliamentary_republic government = prussian_constitutionalism government = hms_government government = constitutional_empire } } } AND = { FROM = { OR = { government = proletarian_dictatorship government = parliamentary_republic government = prussian_constitutionalism } } THIS = { OR = { government = proletarian_dictatorship } } } AND = { FROM = { OR = { government = presidential_dictatorship government = fascist_dictatorship } } THIS = { OR = { government = presidential_dictatorship government = fascist_dictatorship } } } AND = { FROM = { OR = { government = absolute_monarchy government = absolute_empire government = semi_constitutional_empire } } THIS = { OR = { government = absolute_monarchy government = absolute_empire government = semi_constitutional_empire } } } }
			}
			THIS = { set_country_flag = conflicting_government_penalty }
		}
		any_country = {
			limit = { 
				tag = FROM
				badboy = 0.8
			}
			THIS = { set_country_flag = infamous_penalty }
		}
		any_country = {
			limit = { 
				tag = FROM
				have_core_in = THIS
			}
			THIS = { set_country_flag = we_control_core_penalty }
		}
		any_country = {
			limit = { 
				tag = FROM
				OR = {
					has_recently_lost_war = yes
					is_disarmed = yes
					military_access = THIS
				}
			}
			THIS = { set_country_flag = recent_state_bonus }
		}
		any_country = {
			limit = { 
				tag = FROM
				FROM = { slavery = yes_slavery }
				THIS = { slavery = yes_slavery }
			}
			THIS = { set_country_flag = holdout_slavery_bonus }
		}
		any_country = {
			limit = { 
				tag = FROM
				truce_with = THIS
			}
			THIS = { set_country_flag = truce_with_this_penalty }
		}
		set_country_flag = annex_overwrite
    }
	
    option = {
        name = "We shall take them for ourselves."
        any_country = {
			limit = {
				tag = FROM
			}
			set_variable = { which = Annex_Colony value = 1 }
			country_event = 777770630
        }
		change_variable = {
            which = dismantlement_dollars
            value = -1
        }
        set_country_flag = dismantlement_dollars_target badboy = 1
        CLN = { country_event = 777771700 }
        ai_chance = {
            factor = 4
			modifier = { factor = 0.5 has_country_flag = we_control_core_penalty }
			modifier = { factor = 15 badboy = 23 }
			modifier = { factor = 1 badboy = 17 }
			modifier = { factor = 2 OR = { government = hms_government government = constitutional_empire government = absolute_monarchy government = absolute_empire government = semi_constitutional_empire } }
        }
    }
    option = {
        name = "We shall offer $FROMCOUNTRY$ to other victors."
        any_country = {
			limit = {
				tag = FROM
			}
			set_variable = { which = Give_Away_Colony value = 1 }
			country_event = 777770630
        }
        ai_chance = {
            factor = 4
			modifier = { factor = 0.5 has_country_flag = we_control_core_penalty }
			modifier = { factor = 15 badboy = 23 }
			modifier = { factor = 1 badboy = 17 }
			modifier = { factor = 1 OR = { government = hms_government government = constitutional_empire government = absolute_monarchy government = absolute_empire government = semi_constitutional_empire } }
        }
    }
    option = {
        name = "They shall remain with the defeated."
        any_country = {
			limit = {
				tag = FROM
			}
			set_variable = { which = As_Is_Colony value = 1 }
			country_event = 777770630
        }
        ai_chance = {
            factor = 1
			modifier = { factor = 2 has_country_flag = infamous_penalty }
			modifier = { factor = 3 has_country_flag = we_control_core_penalty }
			modifier = { factor = 0.5 has_country_flag = truce_with_this_penalty }
			modifier = { factor = 15 badboy = 23 }
			modifier = { factor = 2 badboy = 17 }
        }
    }
    option = {
        name = "They shall exist as an independent state."
        any_country = {
			limit = {
				tag = FROM
			}
			set_variable = { which = Independent_Colony value = 1 }
			country_event = 777770630
        }
        ai_chance = {
            factor = 2
			modifier = { factor = 15 OR = { government = democracy government = proletarian_dictatorship } }
			modifier = { factor = 2 relation = { who = FROM value = 120 } }
			modifier = { factor = 2 trade_policy = free_trade }
			modifier = { factor = 2 has_country_flag = recent_state_bonus }
			modifier = { factor = 0.5 has_country_flag = we_control_core_penalty }
			modifier = { factor = 0.05 badboy = 23 }
			modifier = { factor = 0.75 badboy = 17 }
        }
    }
